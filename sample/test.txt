script _drop
  changey(0,200,add)
  direction(270)
  repeat(-1)
    speed(rand(140,300))
    fire(drop2, rand(3,6))
    sleep(0.02)
  end
end

script drop2
  changex(0, rand(-200,200), add)
  changespeed($1,rand(-140,-300))
  sleep($1)
end

script _color
  repeat(-1)
    direction(rand(360))
    speed(rand(100,140))
    userdata(rand(4))
    fire()
    sleep(0)
  end
end

script _screw
  fire(screw,$x,$y)
end

script screw
  speed(100)
  repeat(-1)
    changex(0,$1+cos(6*$rep)*30)
    changey(0,$2+sin(6*$rep)*30)
    direction(6*$rep)
    fire()
    sleep(0)
  end
end

script _aim
  repeat(-1)
    fork(aim1)
    sleep(1)
  end
end

script aim1
  direction(-60, aim)
  speed(100)
  repeat(8)
    direction(15, add)
    fire(aim2)
  end
end

script aim2
  changespeed(0.75, 0)
  sleep(0.75)
  direction(0, aim)
  speed(120)
  fire()
  repeat(5)
    speed(18, add)
    fire()
  end
  clear()
end

script _spin
  speed(80)
  repeat(-1)
    direction(180, add)
    fire(spin1)
    direction(160, add)
    fire(spin1)
    sleep(0.075)
  end
end

script spin1
  sleep(rand(1, 2))
  changespeed(1, 0)
  sleep(1)
  changedirection(0, 90)
  changespeed(1, rand(120, 200))
  sleep(1)
end

script _wave
  changey(0,-100,add)
  repeat(-1)

  end
end

script wave1

  direction(160)

    direction(-12, add)

    sleep(0.03)
  end
  sleep(0.5)
  direction(20)
  repeat(12)
    direction(12, add)
    fire(wave2)
    sleep(0.03)
  end
  sleep(0.5)
end
